43  // linear elapsed duration
44  unitDuration = interpElapsed / duration;
45  if (mode == SIN) {

64 void PIDMotor::loop() {
65  calcVel(); // ensure the velocity value is kept up to date
66  if (millis() - lastTimeRunPID > myPID.sampleRateMs) {
67  lastTimeRunPID +=myPID.sampleRateMs;
68  } else {
69  return;
70  }
71  // Done Performing linear/sinusoidal interpolation
72  if (mode != VEL) {
73  // Perform linear/sinusoidal interpolation
74  unitDuration=getInterpolationUnitIncrement();
75  if (unitDuration<1) {
76  float setpointDiff = endSetpoint - startSetpoint;
77  float newSetpoint = startSetpoint + (setpointDiff * unitDuration);
78  Setpoint = newSetpoint;
79  } else {
80  // If there is no interpoation to perform, set the setpoint to the end state
81  Setpoint = endSetpoint;
82  }
83  // Position mode
84  Input = (float) getPosition();
85  //check for integral windup
86  bool thisErrPositive = (Setpoint - Input) > 0;
87  if (thisErrPositive != lastErrPositive) {
88  myPID.clearIntegralBuffer();
89  }
90  lastErrPositive = thisErrPositive;
91  // done with integral windup
92  // run the PID compute function
93  Output = myPID.calc(Setpoint, Input);
94  // use the unit vector function to set the output of the PID
95  setOutputUnitVector(Output);
96  } else {
97  velocityLoop();
98  }
99 }

225 double PIDMotor::calcVel() {
226  //current positions
227  double curPos = getPosition();
228  //current time
229  curTime = millis();
230  //time change in ms from last call
231  timeInterval = curTime - prevTime;
232  if (timeInterval < 50)
233  return Vel; // If this is polled too fast, return the cached velocity
234  //encoder ticks since last call
235  movement = curPos - prevPos;
236  //encoder ticks to degrees
237  double movementdeg = movement / ticksToDeg();
238  //timeInterval in seconds
239  double timeIntervalsec = timeInterval / 1000.0;
240  //Velocity in degrees per seconds
241  Vel = movementdeg / timeIntervalsec;
242  //Serial.println("Vel "+String(movementdeg)+" Per "+String(timeIntervalsec)+" is "+String(Vel)+" max "+String(getFreeSpinMaxDegreesPerSecond()));
243  //sets curent vals to previous
244  prevPos = curPos;
245  prevTime = curTime;
246  return Vel;
