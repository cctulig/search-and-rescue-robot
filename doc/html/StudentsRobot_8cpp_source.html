98 void StudentsRobot::updateStateMachine() {
99  digitalWrite(WII_CONTROLLER_DETECT, 1);
100  long now = millis();
101  switch (status) {
102  case StartupRobot:
103  //Do this once at startup
104  status = StartRunning;
105  Serial.println("StudentsRobot::updateStateMachine StartupRobot here ");
106  break;
107  case StartRunning:
108  Serial.println("Start Running");
109 
110  digitalWrite(H_BRIDGE_ENABLE, 1);
111  // Start an interpolation of the motors
112  motor1->startInterpolationDegrees(motor1->getAngleDegrees(), 1000, SIN);
113  motor2->startInterpolationDegrees(motor2->getAngleDegrees(), 1000, SIN);
114  motor3->startInterpolationDegrees(motor3->getAngleDegrees(), 1000, SIN);
115  status = WAIT_FOR_MOTORS_TO_FINNISH; // set the state machine to wait for the motors to finish
116  nextStatus = Running; // the next status to move to when the motors finish
117  startTime = now + 1000; // the motors should be done in 1000 ms
118  nextTime = startTime + 1000; // the next timer loop should be 1000ms after the motors stop
119  break;
120  case Running:
121  // Set up a non-blocking 1000 ms delay
122  status = WAIT_FOR_TIME;
123  nextTime = nextTime + 100; // ensure no timer drift by incremeting the target
124  // After 1000 ms, come back to this state
125  nextStatus = Running;
126 
127  // Do something
128  if (!digitalRead(BOOT_FLAG_PIN)) {
129  Serial.println(
130  " Running State Machine " + String((now - startTime)));
131 #if defined(USE_IMU)
132  IMU->print();
133 #endif
134 #if defined(USE_IR_CAM)
135  IRCamera->print();
136 #endif
137 
138  }
139  break;
140  case WAIT_FOR_TIME:
141  // Check to see if enough time has elapsed
142  if (nextTime <= millis()) {
143  // if the time is up, move on to the next state
144  status = nextStatus;
145  }
146  break;
147  case WAIT_FOR_MOTORS_TO_FINNISH:
148  if (motor1->isInterpolationDone() && motor2->isInterpolationDone()
149  && motor3->isInterpolationDone()) {
150  status = nextStatus;
151  }
152  break;
153  case Halting:
154  // save state and enter safe mode
155  Serial.println("Halting State machine");
156  digitalWrite(H_BRIDGE_ENABLE, 0);
157  motor3->stop();
158  motor2->stop();
159  motor1->stop();
160 
161  status = Halt;
162  break;
163  case Halt:
164  // in safe mode
165  break;
166 
167  }
168  digitalWrite(WII_CONTROLLER_DETECT, 0);
169 }
